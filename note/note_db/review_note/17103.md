### <br>인터넷 참고 풀이 1 => 시간 초과
```python
t = int(input())

for _ in range(t):
    n = int(input())

    prime_list = [False, False] + [True] * (n - 1)
    for i in range(2, int(n ** 0.5) + 1):
        if prime_list[i]:
            for j in range(i + i, n + 1, i):
                if prime_list[j]:
                    prime_list[j] = False

    count = 0
    for i in range((n // 2) + 1):
        if prime_list[i] and prime_list[n - i]:
            count += 1

    print(count)
```

### <br>인터넷 참고 풀이 2 => 맞았습니다
```python
def get_primary_list(n):
    array = [1 for _ in range(n+1)]

    for i in range(2, int(n ** 0.5) + 1):
        if array[i]:
            j = 2

        while i * j <= n:
            array[i * j] = 0
            j += 1

    return array

T = int(input())
nums = [int(input()) for _ in range(T)]
max_num = max(nums)
primarys = get_primary_list(max_num)

for num in nums:
    result = 0
    for i in range(2, num // 2 + 1):
        if primarys[i] and primarys[num - i]:
            result += 1

    print(result)
```
